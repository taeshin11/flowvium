### A. postProcessPortfolio — ticker 풀 cross-check (line 4506~)
```js
const krSuffixMap = new Map();   // 6자리 → 정확한 .KS or .KQ
for (const t of CANDIDATE_TICKERS) {
  const m = t.match(/^(\d{6})\.(KS|KQ)$/);
  if (m) krSuffixMap.set(m[1], t);
}
// 6자리 코드 → 풀에서 정확한 suffix 찾기
if (KR_NUM.test(ticker)) ticker = krSuffixMap.get(ticker) ?? `${ticker}.KS`;
// 잘못된 suffix swap (.KS↔.KQ)
// 풀에 없는 6자리 코드 → reject (LLM 환각 차단)
```
