### 2. Post-parse 가드 (hard fallback)
```typescript
function hasChineseLeak(text: string): boolean {
  return /[\u4e00-\u9fff]/.test(text);
}
// parseCascade 내:
if (typeof summary === 'string' && hasChineseLeak(summary)) {
  logger.warn('news-cascade', 'chinese_leak_summary', { link, sample: summary.slice(0,80) });

}
```
